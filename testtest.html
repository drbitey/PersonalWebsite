const map = new maplibregl.Map({ container: 'map', style: 'https://basemaps.cartocdn.com/gl/dark-matter-gl-style/style.json', center: [0, 20], // Center on Atlantic to see both sides zoom: 2 }); let currentDecade = null; // Helper function to load a CSV and turn it into GeoJSON function loadDecadeData(decade) { const filePath = `data_chunks/music_${decade}s.csv`; Papa.parse(filePath, { download: true, header: true, dynamicTyping: true, complete: function(results) { // Convert CSV rows to GeoJSON Features const features = results.data .filter(row => row.lat && row.lon) // Clean any empty rows .map(row => ({ 'type': 'Feature', 'geometry': { 'type': 'Point', 'coordinates': [row.lon, row.lat] }, 'properties': { ...row } })); const geojson = { 'type': 'FeatureCollection', 'features': features }; // Update the map source if (map.getSource('music-data')) { map.getSource('music-data').setData(geojson); } else { map.addSource('music-data', { 'type': 'geojson', 'data': geojson }); addHeatmapLayer(); } } }); } function addHeatmapLayer() { map.addLayer({ 'id': 'music-heat', 'type': 'heatmap', 'source': 'music-data', 'paint': { 'heatmap-weight': 0.1, 'heatmap-intensity': 1, 'heatmap-radius': [['interpolate', ['linear'], ['zoom'], 0, 2, 9, 15]], 'heatmap-color': [ 'interpolate', ['linear'], ['heatmap-density'], 0, 'rgba(0,0,0,0)', 0.2, '#4d004b', 0.4, '#810f7c', 0.6, '#ba478f', 0.8, '#df65b0', 1, '#f7fcfd' ] } }); } map.on('load', () => { // Load initial decade (1980s) currentDecade = 1980; loadDecadeData(currentDecade); const slider = document.getElementById('slider'); const yearLabel = document.getElementById('year-label'); slider.addEventListener('input', (e) => { const year = parseInt(e.target.value); const decade = Math.floor(year / 10) * 10; yearLabel.textContent = year; // IF THE DECADE CHANGED, FETCH NEW DATA if (decade !== currentDecade) { currentDecade = decade; loadDecadeData(decade); console.log(`Swapping to ${decade}s data...`); } // Filter points to only show releases for the exact year // Note: We use '<=' to show the 'build up' of music over time map.setFilter('music-heat', ['<=', ['get', 'release_year'], year]); }); });
